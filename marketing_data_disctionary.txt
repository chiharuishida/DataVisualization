ID = Customer's unique identifier
Year_Birth = Customer’s birth year
Education = Customer’s education level
Marital Status = Customer’s marital status
Income = Customer’s household annual income
Kidhome = number of children in customer’s household
Teenhome = number of teenagers in customer’s household
Dt_Customer = Date of customer’s enrollment with the company
Recency = Number od days since customer’s last purchase
MintWines = Amount spent on wine in the last 2 years
MintFruits = Amount spent on fruits in the last 2 years
MintMeatProducts = Amount spent on meat in the last 2 years
MintFishProducts = Amount spent on fish in the last 2 years
MintSweetProducts = Amount spent on sweets in the last 2 years
MintGoldProducts = Amount on gold in the last 2 years
NumDealsPurchase = Number of purchases made with a discount
NumWebPurchases = Number of purchases made through the company’s web site
NumStorePurchase = Number of purchases made directly in stores
NumWebVisitsMo = Number of visits to company’s website in the last month
AcceptedCmp4 = 1 if customer accepted the offer in the 4th campaign. 0 otherwise
AcceptedCmp5 = 1 if customer accepted the offer in the 5th campaign. 0 otherwise
AcceptedCmp1 = 1 if customer accepted the offer in the 1st campaign. 0 otherwise
AcceptedCmp2 = 1 if customer accepted the offer in the 2nd campaign. 0 otherwise
Response = 1 if customer accepted the offer in the last campaign, 0 otherwise
Complain = 1 if customer complained in the last 2 years. 0 otherwise
